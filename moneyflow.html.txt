<script>
const MONTHS = ["มกราคม","กุมภาพันธ์","มีนาคม","เมษายน","พฤษภาคม","มิถุนายน","กรกฎาคม","สิงหาคม","กันยายน","ตุลาคม","พฤศจิกายน","ธันวาคม"];
const CAT_EXP = ["อาหาร","เดินทาง","ที่พัก","สุขภาพ","บันเทิง","ช้อปปิ้ง","ค่าสาธารณูปโภค","การศึกษา","อื่นๆ"];
const CAT_INC = ["เงินเดือน","ฟรีแลนซ์","ดอกเบี้ย/ปันผล","ขายของ","โบนัส","อื่นๆ"];

const now = new Date();
let selMonth = now.getMonth(), selYear = now.getFullYear();
let activeTab = "overview", currentType = "expense";
let txData = [];
let deferredPrompt = null;

try { txData = JSON.parse(localStorage.getItem("mf_tx") || "[]"); } catch {}
function save() { try { localStorage.setItem("mf_tx", JSON.stringify(txData)); } catch {} }
function fmt(n) { return n.toLocaleString("th-TH", {minimumFractionDigits:2, maximumFractionDigits:2}); }

// Month/Year selects
(function() {
  const sm = document.getElementById("sel-month");
  MONTHS.forEach((m,i) => { const o=document.createElement("option"); o.value=i; o.textContent=m; if(i===selMonth)o.selected=true; sm.appendChild(o); });
  const sy = document.getElementById("sel-year");
  for(let y=selYear-3;y<=selYear+2;y++) { const o=document.createElement("option"); o.value=y; o.textContent=(y+543)+""; if(y===selYear)o.selected=true; sy.appendChild(o); }
  sm.onchange = () => { selMonth=+sm.value; render(); };
  sy.onchange = () => { selYear=+sy.value; render(); };
  document.getElementById("inp-date").value = now.toISOString().slice(0,10);
})();

// Tabs

  btn.onclick = () => {
    document.querySelectorAll(".tab").forEach(b=>b.classList.remove("active"));
    btn.classList.add("active");
    activeTab = btn.dataset.tab;
    render();
  };
});

// Add
document.getElementById("submit-btn").onclick = () => {
  const amt = parseFloat(document.getElementById("inp-amount").value);
  const cat = document.getElementById("inp-category").value;
  const date = document.getElementById("inp-date").value || now.toISOString().slice(0,10);
  const note = document.getElementById("inp-note").value.trim();
  if(!amt||!cat) { (amt?document.getElementById("inp-category"):document.getElementById("inp-amount")).focus(); return; }
  txData.unshift({id:Date.now(),type:currentType,amount:amt,category:cat,date,note});
  save();
  document.getElementById("inp-amount").value="";
  document.getElementById("inp-category").value="";
  document.getElementById("inp-note").value="";
  closeModal();
  render();

// Delete
window.deleteTx = function(id) {
  if(!confirm("ลบรายการนี้?")) return;
  txData = txData.filter(t=>t.id!==id);
  save(); render();
};

function getFiltered() {
  return txData.filter(t=>{ const d=new Date(t.date); return d.getMonth()===selMonth&&d.getFullYear()===selYear; });
}

function render() {
  const f = getFiltered();
  const income = f.filter(t=>t.type==="income").reduce((s,t)=>s+t.amount,0);
  const expense = f.filter(t=>t.type==="expense").reduce((s,t)=>s+t.amount,0);
  const balance = income-expense;
  const bEl = document.getElementById("balance-amt");
  bEl.textContent = (balance>=0?"+":"")+fmt(balance)+" ฿";
  bEl.className = "balance-amount "+(balance>=0?"pos":"neg");
  document.getElementById("total-income").textContent = "฿ "+fmt(income);
  document.getElementById("total-expense").textContent = "฿ "+fmt(expense);

  const el = document.getElementById("tab-content");
  if(activeTab==="overview") {
    const pct = income>0?Math.min(expense/income*100,100):0;
    const over = income>0&&expense/income>0.8;
    let h = `<div class="prog-card">
      <div class="sec-label" style="margin-bottom:10px">สัดส่วนการใช้จ่าย</div>
      <div class="prog-row"><span style="color:var(--muted)">รายจ่าย / รายรับ</span><span style="color:${over?"var(--red)":"var(--green)"};font-weight:700">${income>0?Math.round(pct):"-"}%</span></div>
      <div class="prog-track"><div class="prog-fill" style="width:${pct}%;background:${over?"linear-gradient(90deg,#f87171,#ef4444)":"linear-gradient(90deg,#4ade80,#22c55e)"}"></div></div>
      <div class="prog-note">${income===0?"ยังไม่มีรายรับในเดือนนี้":over?"⚠️ ใช้จ่ายเกิน 80% ของรายรับแล้ว":"✅ การใช้จ่ายอยู่ในระดับปกติ"}</div>
    </div>`;
    h += `<div class="sec-label">รายการล่าสุด</div>`;
    const rec = f.slice().sort((a,b)=>new Date(b.date)-new Date(a.date)).slice(0,6);
    if(!rec.length) h += empty("📝","ยังไม่มีรายการ","กดปุ่ม + เพื่อเพิ่มรายการ");
    else rec.forEach(t=>h+=txHTML(t,false));
    el.innerHTML=h;
  } else if(activeTab==="transactions") {
    let h = `<div class="sec-label">รายการทั้งหมด ${f.length} รายการ</div>`;
    const s = f.slice().sort((a,b)=>new Date(b.date)-new Date(a.date));
    if(!s.length) h+=empty("📋","ไม่มีรายการ","กดปุ่ม + เพื่อเพิ่มรายการ");
    else s.forEach(t=>h+=txHTML(t,true));
    el.innerHTML=h;
  } else {
    let h = `<div class="sec-label">หมวดหมู่รายจ่าย</div>`;
    const em={}; f.filter(t=>t.type==="expense").forEach(t=>{em[t.category]=(em[t.category]||0)+t.amount;});
    const ee=Object.entries(em).sort((a,b)=>b[1]-a[1]);
    if(!ee.length) h+=empty("💸","ยังไม่มีรายจ่าย","");
    else ee.forEach(([c,a])=>{ const p=expense>0?Math.round(a/expense*100):0; h+=`<div class="cat-bar"><div class="cat-row"><span>${c}</span><span style="color:var(--red);font-weight:700">${fmt(a)} ฿</span></div><div class="prog-track"><div class="prog-fill" style="width:${p}%;background:linear-gradient(90deg,#f87171,#a855f7)"></div></div><div class="cat-pct">${p}% ของรายจ่ายทั้งหมด</div></div>`; });
    const im={}; f.filter(t=>t.type==="income").forEach(t=>{im[t.category]=(im[t.category]||0)+t.amount;});
    const ie=Object.entries(im).sort((a,b)=>b[1]-a[1]);
    if(ie.length){ h+=`<div class="sec-label" style="margin-top:20px">หมวดหมู่รายรับ</div>`; ie.forEach(([c,a])=>{ const p=income>0?Math.round(a/income*100):0; h+=`<div class="cat-bar"><div class="cat-row"><span>${c}</span><span style="color:var(--green);font-weight:700">${fmt(a)} ฿</span></div><div class="prog-track"><div class="prog-fill" style="width:${p}%;background:linear-gradient(90deg,#4ade80,#22c55e)"></div></div></div>`; }); }
    el.innerHTML=h;
  }
}

function txHTML(t, showDate) {
  const inc=t.type==="income";
  const d=showDate?`<div class="tx-note">${new Date(t.date).toLocaleDateString("th-TH",{day:"numeric",month:"short",year:"numeric"})}</div>`:"";
  const n=t.note?`<div class="tx-note">${t.note}</div>`:"";
  return `<div class="tx"><div class="tx-icon ${inc?"inc":"exp"}">${inc?"💵":"💸"}</div><div class="tx-info"><div class="tx-cat">${t.category}</div>${n}${d}</div><div class="tx-amt ${inc?"inc":"exp"}">${inc?"+":"-"}${fmt(t.amount)} ฿</div><button class="tx-del" onclick="deleteTx(${t.id})">🗑</button></div>`;
}

// ── EXPORT JSON ──
function exportJSON() {
  if(!txData.length) { showToast("ไม่มีข้อมูลให้ส่งออก", true); return; }
  const payload = { version: 1, exportedAt: new Date().toISOString(), transactions: txData };
  const blob = new Blob([JSON.stringify(payload, null, 2)], {type:"application/json"});
  const url = URL.createObjectURL(blob);
  const a = document.createElement("a");
  const dateStr = new Date().toISOString().slice(0,10);
  a.href = url; a.download = "moneyflow-" + dateStr + ".json";
  a.click(); URL.revokeObjectURL(url);
  showToast("✅ ส่งออกข้อมูลสำเร็จ!");
}

// ── IMPORT JSON ──
document.getElementById("file-input").onchange = function(e) {
  const file = e.target.files[0];
  if(!file) return;
  const reader = new FileReader();
  reader.onload = function(ev) {
    try {
      const parsed = JSON.parse(ev.target.result);
      let imported = [];
      if(Array.isArray(parsed)) imported = parsed;
      else if(parsed.transactions && Array.isArray(parsed.transactions)) imported = parsed.transactions;
      else { showToast("ไฟล์ไม่ถูกต้อง", true); return; }
      if(!imported.length) { showToast("ไม่พบรายการในไฟล์", true); return; }
      const existIds = new Set(txData.map(t=>t.id));
      const newItems = imported.filter(t=>!existIds.has(t.id));
      if(!newItems.length) { showToast("ข้อมูลซ้ำ ไม่มีรายการใหม่", true); return; }
      if(confirm("พบ " + newItems.length + " รายการใหม่\nนำเข้าเพิ่มเติม (ไม่ลบข้อมูลเดิม)?")) {
        txData = [...txData, ...newItems].sort((a,b)=>b.id-a.id);
        save(); render();
        showToast("✅ นำเข้า " + newItems.length + " รายการสำเร็จ!");
      }
    } catch(err) { showToast("เปิดไฟล์ไม่ได้ กรุณาตรวจสอบ", true); }
    e.target.value = "";
  };
  reader.readAsText(file);
};

// ── TOAST ──
function showToast(msg, isError) {
  const t = document.getElementById("toast");
  t.textContent = msg;
  t.className = "show" + (isError ? " error" : "");
  clearTimeout(t._timer);
  t._timer = setTimeout(() => { t.className = ""; }, 2800);
}
